const partial = (fn, ...partialArgs) => {
  let args = partialArgs;
	
  return function(...restArgs) {
	let argIndex = 0;
	for(let i = 0;i<args.length && argIndex<restArgs.length;i+=1) {
	  if(args[i] === undefined) args[i] = restArgs[argIndex++]
	}
	  
    return fn.apply(null, args)
  }
}
